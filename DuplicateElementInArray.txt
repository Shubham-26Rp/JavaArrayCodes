public class Main
{
	public static void main(String[] args) {
	  int arr[] =new int[]{1,2,3,4,6,7,8,8,9};
	  for(int i=0;i<arr.length;i++)
	  {
	      for(int j=i+1;j<arr.length;j++){
	          if(arr[i]==arr[j])
	          {
	              arr[i]=arr[j];
	              System.out.println(arr[i]);
	          }
	      }
	  }
	}
}
